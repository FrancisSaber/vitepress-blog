var findAndReplacePattern = function(words, pattern) {
    let toNum = (word)=>{
        let arr =[],map={},t=0
        for(let i of word){
            if(!map[i]){
                t++
                map[i] = t
            }
            arr.push(map[i])
        }
        return arr.toString()
    }
    let pa = toNum(pattern),res=[]
    for(let i of words){
        if(toNum(i)===pa){
            res.push(i)
        }
    }
    return res
};
